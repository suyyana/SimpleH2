Install

Example

HTTP2

const simpleH2 = require('simpleH2');
const srv = new simpleH2.Server();
srv.get('/', (stream, headers, params, next) => {
  next('Hello, world!');
});
srv.listen(3001);

HTTP2 With SSL

const fs = require('fs');
const simpleH2 = require('simpleH2');
const srv = new simpleH2.Server({
  key: fs.readFileSync('key.private.pem', 'utf8'),
  cert: fs.readFileSync('cert.pem', 'utf8'),
});
srv.get('/', (stream, headers, params, next) => {
  next('Hello, world!');
});
srv.listen(3001);
